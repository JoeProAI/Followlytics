## Architecture Components

### Coordinator Service

```python
from daytona import Daytona, CreateSandboxFromSnapshotParams, Resources

class ScanCoordinator:
    def __init__(self, daytona_client):
        self.daytona = daytona_client
        self.workers = []
        self.progress = {}
    
    async def create_coordinator(self):
        """Create the main coordinator sandbox"""
        return await self.daytona.create(
            CreateSandboxFromSnapshotParams(
                snapshot="followlytics-coordinator",
                resources=Resources(cpu=2, memory=4, disk=20),
                auto_stop_interval=0,  # Never auto-stop
                volumes=[VolumeMount("shared-data", "/coordinator/data")]
            )
        )
    
    async def scale_workers(self, follower_count):
        """Auto-scale workers based on follower count"""
        worker_count = min(50, max(1, follower_count // 20000))
        
        for i in range(worker_count):
            worker = await self.daytona.create(
                CreateSandboxFromSnapshotParams(
                    snapshot="followlytics-worker",
                    resources=Resources(cpu=4, memory=8, disk=50),
                    auto_stop_interval=30,  # Auto-stop after 30 min idle
                    volumes=[VolumeMount("shared-data", f"/worker-{i}/data")],
                    env_vars={
                        "WORKER_ID": str(i),
                        "COORDINATOR_ID": self.coordinator.id,
                        "CHUNK_SIZE": "10000"
                    }
                )
            )
            self.workers.append(worker)
```

### Worker Fleet Management

```python
class WorkerFleet:
    def __init__(self, coordinator):
        self.coordinator = coordinator
        self.active_workers = {}
        self.job_queue = []
    
    async def distribute_work(self, target_username, total_followers):
        """Distribute follower scanning across workers"""
        chunk_size = 10000
        chunks = []
        
        for start in range(0, total_followers, chunk_size):
            end = min(start + chunk_size, total_followers)
            chunks.append({
                "start": start,
                "end": end,
                "username": target_username,
                "chunk_id": f"{target_username}_{start}_{end}"
            })
        
        # Distribute chunks to available workers
        for i, chunk in enumerate(chunks):
            worker_id = i % len(self.workers)
            await self.assign_chunk_to_worker(worker_id, chunk)
    
    async def monitor_progress(self):
        """Real-time progress monitoring"""
        while self.has_active_jobs():
            for worker in self.workers:
                progress = await worker.get_progress()
                self.update_progress(worker.id, progress)
            
            await asyncio.sleep(5)  # Check every 5 seconds
```

### Intelligent Load Balancing

<CodeGroup>
```python Python
class LoadBalancer:
    def __init__(self):
        self.worker_loads = {}
        self.rate_limits = {}
    
    def assign_optimal_worker(self, chunk):
        """Assign chunk to least loaded worker"""
        available_workers = [
            w for w in self.workers 
            if not self.is_rate_limited(w.id)
        ]
        
        if not available_workers:
            return self.wait_for_rate_limit_reset()
        
        # Find worker with lowest current load
        optimal_worker = min(
            available_workers,
            key=lambda w: self.worker_loads.get(w.id, 0)
        )
        
        return optimal_worker
```

```typescript TypeScript
class LoadBalancer {
    private workerLoads: Map<string, number> = new Map();
    private rateLimits: Map<string, Date> = new Map();
    
    assignOptimalWorker(chunk: ScanChunk): Worker {
        const availableWorkers = this.workers.filter(
            w => !this.isRateLimited(w.id)
        );
        
        if (availableWorkers.length === 0) {
            return this.waitForRateLimitReset();
        }
        
        // Find worker with lowest current load
        return availableWorkers.reduce((optimal, current) => {
            const currentLoad = this.workerLoads.get(current.id) || 0;
            const optimalLoad = this.workerLoads.get(optimal.id) || 0;
            return currentLoad < optimalLoad ? current : optimal;
        });
    }
}
```

### Rate Limit Management

```python
class RateLimitManager:
    def __init__(self):
        self.limits = {
            "requests_per_minute": 300,
            "requests_per_hour": 15000,
            "concurrent_connections": 50
        }
        self.usage_tracking = {}
    
    async def throttle_worker(self, worker_id):
        """Implement intelligent throttling"""
        current_usage = self.get_worker_usage(worker_id)
        
        if current_usage > self.limits["requests_per_minute"]:
            delay = self.calculate_backoff_delay(current_usage)
            await asyncio.sleep(delay)
        
        return True
    
    def calculate_backoff_delay(self, usage_rate):
        """Exponential backoff with jitter"""
        base_delay = min(60, (usage_rate / 100) ** 2)
        jitter = random.uniform(0.8, 1.2)
        return base_delay * jitter
```

```python
class DataSynchronizer:
    def __init__(self, volume_path):
        self.volume_path = volume_path
        self.sync_interval = 10  # seconds
    
    async def sync_worker_data(self, worker_id, chunk_data):
        """Sync worker data to shared volume"""
        chunk_file = f"{self.volume_path}/chunks/{worker_id}_chunk.json"
        
        # Atomic write with backup
        temp_file = f"{chunk_file}.tmp"
        with open(temp_file, 'w') as f:
            json.dump(chunk_data, f)
        
        os.rename(temp_file, chunk_file)
        
        # Update progress tracking
        await self.update_progress_tracker(worker_id, len(chunk_data))
    
    async def aggregate_results(self):
        """Combine all worker results"""
        all_followers = []
        chunk_files = glob.glob(f"{self.volume_path}/chunks/*_chunk.json")
        
        for chunk_file in chunk_files:
            with open(chunk_file, 'r') as f:
                chunk_data = json.load(f)
                all_followers.extend(chunk_data)
        
        # Deduplicate and validate
        unique_followers = self.deduplicate_followers(all_followers)
        return self.validate_data_integrity(unique_followers)
```

```python
class PerformanceMonitor:
    def __init__(self):
        self.metrics = {
            "throughput": [],
            "error_rates": [],
            "resource_usage": [],
            "cost_tracking": []

    async def collect_metrics(self):

            self.metrics["throughput"].append({
                "worker_id": worker.id,
                "followers_per_minute": metrics.followers_processed / metrics.runtime_minutes,
                "timestamp": datetime.utcnow()
            })
            
            self.metrics["resource_usage"].append({
                "worker_id": worker.id,
                "cpu_usage": metrics.cpu_percent,
                "memory_usage": metrics.memory_mb,
                "network_io": metrics.network_bytes
            })
    
    def generate_cost_report(self):
        """Calculate real-time cost breakdown"""
        total_cost = 0
        
        for worker in self.workers:
            runtime_hours = worker.get_runtime_hours()
            worker_cost = runtime_hours * worker.hourly_rate
            total_cost += worker_cost
        
        return {
            "total_cost": total_cost,
            "cost_per_follower": total_cost / self.total_followers_processed,
            "estimated_completion_cost": self.project_final_cost()
        }
```

## Failure Recovery

### Checkpoint System

```python
class CheckpointManager:
    def __init__(self, volume_path):
        self.checkpoint_path = f"{volume_path}/checkpoints"
        self.checkpoint_interval = 300  # 5 minutes
    
    async def create_checkpoint(self, scan_state):
        """Create recovery checkpoint"""
        checkpoint = {
            "timestamp": datetime.utcnow().isoformat(),
            "total_followers": scan_state.total_followers,
            "processed_followers": scan_state.processed_followers,
            "worker_states": {
                w.id: w.get_state() for w in scan_state.workers
            },
            "completed_chunks": scan_state.completed_chunks,
            "failed_chunks": scan_state.failed_chunks
        }
        
        checkpoint_file = f"{self.checkpoint_path}/checkpoint_{int(time.time())}.json"
        with open(checkpoint_file, 'w') as f:
            json.dump(checkpoint, f)
    
    async def restore_from_checkpoint(self, checkpoint_file):
        """Resume scan from checkpoint"""
        with open(checkpoint_file, 'r') as f:
            checkpoint = json.load(f)
        
        # Recreate worker fleet
        await self.recreate_workers(checkpoint["worker_states"])
        
        # Resume from last processed position
        remaining_chunks = self.calculate_remaining_work(checkpoint)
        await self.distribute_remaining_work(remaining_chunks)
```

## Cost Optimization

### Dynamic Resource Scaling

```python
class CostOptimizer:
    def __init__(self):
        self.cost_targets = {
            "max_cost_per_follower": 0.001,  # $0.001 per follower
            "max_total_cost": 500,           # $500 max per scan
            "efficiency_threshold": 0.8      # 80% resource utilization
        }
    
    async def optimize_worker_allocation(self, current_progress):
        """Dynamically adjust worker count for cost efficiency"""
        efficiency = self.calculate_efficiency(current_progress)
        
        if efficiency < self.cost_targets["efficiency_threshold"]:
            # Scale down underutilized workers
            await self.scale_down_workers()
        
        elif self.is_behind_schedule(current_progress):
            # Scale up if behind schedule and within budget
            if self.projected_cost() < self.cost_targets["max_total_cost"]:
                await self.scale_up_workers()
    
    def calculate_roi_metrics(self):
        """Calculate return on investment metrics"""
        return {
            "cost_per_follower": self.total_cost / self.followers_processed,
            "time_to_completion": self.estimate_completion_time(),
            "efficiency_score": self.calculate_efficiency_score(),
            "cost_savings_vs_traditional": self.calculate_savings()
        }
```
